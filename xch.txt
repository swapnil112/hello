from transformers import AutoTokenizer, AutoModelForSeq2SeqLM

# Load ClinicalBERT tokenizer and model
tokenizer = AutoTokenizer.from_pretrained("emilyalsentzer/Bio_ClinicalBERT")
model = AutoModelForSeq2SeqLM.from_pretrained("emilyalsentzer/Bio_ClinicalBERT")

# Define the text to be summarized
text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Suspendisse vel felis mauris. Quisque ut mauris tincidunt, dictum sapien vel, blandit lectus. Sed lacinia aliquet odio, ac ullamcorper ipsum lacinia vel. Fusce eu est suscipit, congue tellus vel, consequat elit. Nullam ut justo vel risus sodales semper nec et libero."

# Tokenize the text
inputs = tokenizer.encode(text, return_tensors="pt")

# Generate summary
summary_ids = model.generate(inputs, num_beams=4, max_length=50, early_stopping=True)
summary = tokenizer.decode(summary_ids[0], skip_special_tokens=True)

print(summary)
